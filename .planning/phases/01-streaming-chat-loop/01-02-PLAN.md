<interfaces>
<!-- API contract from Plan 01 (Go backend) -->
POST /api/chat/stream
Request body: { "sessionId": string, "message": string }
Response: SSE stream — each event is "data: TOKEN\n\n"
          Error event: "data: [ERROR] message\n\n"
          Stream ends when the HTTP connection closes (io.EOF)

<!-- Session contract -->
Session ID: UUID stored in sessionStorage under key "jarvis-session-id"
Resets on tab close, survives same-tab navigation.

<!-- streamChat signature (frontend/lib/api.ts) -->
export async function streamChat(
  sessionId: string,
  message: string,
  onToken: (token: string) => void,
  onDone: () => void,
  onError: (err: Error) => void,
): Promise<void>

<!-- Message type (frontend/app/page.tsx) -->
type Message = {
  id: string
  role: 'user' | 'assistant'
  content: string
  isStreaming?: boolean
}
