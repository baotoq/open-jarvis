```go
func ListConversationsHandler(svcCtx *svc.ServiceContext) http.HandlerFunc {
    return func(w http.ResponseWriter, r *http.Request) {
        l := logic.NewListConvsLogic(r.Context(), svcCtx)
        convs, err := l.ListConversations()
        if err != nil {
            logx.WithContext(r.Context()).Errorf("...")
            http.Error(w, "internal server error", http.StatusInternalServerError)
            return
        }
        w.Header().Set("Content-Type", "application/json")
        json.NewEncoder(w).Encode(convs)
    }
}
```

    ```go
    type SearchConvsLogic struct {
        ctx    context.Context
        svcCtx *svc.ServiceContext
    }
    func NewSearchConvsLogic(ctx context.Context, svcCtx *svc.ServiceContext) *SearchConvsLogic
    func (l *SearchConvsLogic) Search(query string) ([]types.SearchResult, error) {
        // Type-assert svcCtx.Store to ConvSearcher; if not supported return empty
        searcher, ok := l.svcCtx.Store.(ConvSearcher)
        if !ok {
            return nil, nil
        }
        results, err := searcher.SearchConversations(query)
        if err != nil {
            return nil, err
        }
        // Map svc.SearchResult → types.SearchResult
        out := make([]types.SearchResult, len(results))
        for i, r := range results {
            out[i] = types.SearchResult{
                ID:        r.ConversationID,
                Title:     r.Title,
                UpdatedAt: r.UpdatedAt,
                Snippet:   r.Snippet,
            }
        }
        return out, nil
    }
    ```
